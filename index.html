/** * GAJI PAYROLL — Google Sheets backend (Apps Script) * ---------------------------------------------------- * This turns a Google Sheet into a simple database for the payroll * system, so the HTML file hosted on GitHub Pages can read/write to it. * * SETUP: * 1. Go to https://sheets.google.com and create a new blank spreadsheet. * Rename it e.g. "Gaji Payroll Database". * 2. In the sheet, go to Extensions > Apps Script. * 3. Delete any starter code, paste this whole file in, and save * (name the project e.g. "Gaji Backend"). * 4. Click Deploy > New deployment. * - Type: "Web app" * - Description: anything * - Execute as: Me * - Who has access: Anyone (needed so the GitHub-hosted page can call it) * 5. Click Deploy, authorize the permissions Google asks for. * 6. Copy the "Web app URL" it gives you (looks like * https://script.google.com/macros/s/XXXXXXXX/exec). * 7. Paste that URL into the payslip-malaysia.html file where it says * const API_URL = "https://script.google.com/macros/s/AKfycbw9GRBWrA5jl8RNhLHRSmXMeZhSt1Fz4U0neV-afpZB7yrCWHMMiU7Ed16THUdyVl-g/exec"; * 8. Push the HTML file to a GitHub repo and enable GitHub Pages * (Settings > Pages > Deploy from branch > main / root). * * This script auto-creates 3 sheet tabs the first time it runs: * Companies | Staff | Payslips */ function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { const params = e.parameter; const action = params.action; const body = e.postData ? JSON.parse(e.postData.contents || '{}') : {}; let result; switch (action) { case 'listCompanies': result = listRows('Companies'); break; case 'saveCompany': result = upsertRow('Companies', body); break; case 'deleteCompany': result = deleteRow('Companies', body.id); cascadeDeleteCompany(body.id); break; case 'listStaff': result = listRows('Staff', 'companyId', params.companyId); break; case 'saveStaff': result = upsertRow('Staff', body); break; case 'deleteStaff': result = deleteRow('Staff', body.id); cascadeDeleteStaff(body.id); break; case 'listPayslips': result = listRows('Payslips', 'staffId', params.staffId); break; case 'getPayslip': result = getPayslip(params.staffId, params.year, params.month, params.variant); break; case 'savePayslip': result = upsertPayslip(body); break; default: result = { error: 'Unknown action: ' + action }; } return jsonOut(result); } catch (err) { return jsonOut({ error: err.message }); } } function jsonOut(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); const headers = name === 'Companies' ? ['id','companyName','companyAddr','ssmNo','companyRefs','hrdfRegistered'] : name === 'Staff' ? ['id','companyId','empName','empNo','empIC','empPosition','empDept','empStatus','empAge','empEpfNo','empSocsoNo','empTaxNo'] : ['id','companyId','staffId','year','month','variant','data']; // Payslips sheet.appendRow(headers); } return sheet; } function listRows(sheetName, filterKey, filterValue) { const sheet = getSheet(sheetName); const values = sheet.getDataRange().getValues(); const headers = values[0]; const rows = []; for (let i = 1; i < values.length; i++) { const row = {}; headers.forEach((h, idx) => row[h] = values[i][idx]); if (!filterKey || String(row[filterKey]) === String(filterValue)) rows.push(row); } return rows; } function upsertRow(sheetName, data) { const sheet = getSheet(sheetName); const values = sheet.getDataRange().getValues(); const headers = values[0]; if (!data.id) data.id = sheetName.toLowerCase() + '_' + new Date().getTime(); let rowIndex = -1; for (let i = 1; i < values.length; i++) { if (String(values[i][0]) === String(data.id)) { rowIndex = i + 1; break; } } const rowValues = headers.map(h => data[h] !== undefined ? data[h] : ''); if (rowIndex > 0) sheet.getRange(rowIndex, 1, 1, headers.length).setValues([rowValues]); else sheet.appendRow(rowValues); return data; } function deleteRow(sheetName, id) { const sheet = getSheet(sheetName); const values = sheet.getDataRange().getValues(); for (let i = 1; i < values.length; i++) { if (String(values[i][0]) === String(id)) { sheet.deleteRow(i + 1); break; } } return { deleted: id }; } function cascadeDeleteCompany(companyId) { const staff = listRows('Staff', 'companyId', companyId); staff.forEach(s => { deleteRow('Staff', s.id); cascadeDeleteStaff(s.id); }); } function cascadeDeleteStaff(staffId) { const slips = listRows('Payslips', 'staffId', staffId); slips.forEach(p => deleteRow('Payslips', p.id)); } function getPayslip(staffId, year, month, variant) { const v = variant || 'A'; const rows = listRows('Payslips', 'staffId', staffId); const match = rows.find(r => String(r.year) === String(year) && String(r.month) === String(month) && String(r.variant || 'A') === String(v)); if (!match) return null; return Object.assign({}, match, { data: JSON.parse(match.data || '{}') }); } function upsertPayslip(body) { // body: { companyId, staffId, year, month, variant, data: {...} } const v = body.variant || 'A'; const rows = listRows('Payslips', 'staffId', body.staffId); const existing = rows.find(r => String(r.year) === String(body.year) && String(r.month) === String(body.month) && String(r.variant || 'A') === String(v)); const record = { id: existing ? existing.id : 'pay_' + new Date().getTime(), companyId: body.companyId, staffId: body.staffId, year: body.year, month: body.month, variant: v, data: JSON.stringify(body.data) }; return upsertRow('Payslips', record); }
